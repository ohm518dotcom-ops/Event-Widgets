    async function fetchEvents() {

            const response = await fetch(`${SUPABASE_URL}/rest/v1/events?select=*&start_time=gte.${startStr}&start_time=lte.${endStr}`, {

    function renderCalendar() {
        grid.innerHTML = '';
        const year = currentDate.getFullYear();
        const month = currentDate.getMonth();
        
        const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
        monthYear.textContent = `${monthNames[month]} ${year}`;

        const firstDay = new Date(year, month, 1).getDay();
        const daysInMonth = new Date(year, month + 1, 0).getDate();
        const daysInPrevMonth = new Date(year, month, 0).getDate();
        const today = new Date();
        const isCurrentMonth = today.getMonth() === month && today.getFullYear() === year;

        // Populate days
        for (let i = 0; i < 42; i++) {
            const dayEl = document.createElement('div');
            dayEl.className = 'ohm-day';

            if (i < firstDay) {
                dayEl.textContent = daysInPrevMonth - firstDay + i + 1;
                dayEl.classList.add('other-month');
            } else if (i >= firstDay && i < firstDay + daysInMonth) {
                const dayNum = i - firstDay + 1;
                dayEl.textContent = dayNum;
                dayEl.classList.add('active-month');
                
                if (isCurrentMonth && dayNum === today.getDate()) {
                    dayEl.classList.add('today');
                }

                // Check events
                const cellDate = new Date(year, month, dayNum);
                dayEl.dataset.date = `${year}-${month}-${dayNum}`;
                const dayEvents = events.filter(e => 
                    e.start_time.getFullYear() === year &&
                    e.start_time.getMonth() === month &&
                    e.start_time.getDate() === dayNum
                );
                
                if (dayEvents.length > 0) {
                    dayEl.classList.add('has-events');
                    
                    const formatterDate = new Intl.DateTimeFormat('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
                    dayEl.addEventListener('click', () => selectDay(dayEl, formatterDate.format(cellDate), dayEvents));
                }
            } else {
                dayEl.textContent = i - firstDay - daysInMonth + 1;
                dayEl.classList.add('other-month');
            }
            grid.appendChild(dayEl);
        }
        function selectUpcomingEvent() {
        const now = new Date();
        const upcomingEvents = [...events].filter(e => e.start_time >= now).sort((a,b) => a.start_time - b.start_time);
        
        if (upcomingEvents.length > 0) {
            const nextEvent = upcomingEvents[0];
            const dayEvents = events.filter(e => 
                e.start_time.getFullYear() === nextEvent.start_time.getFullYear() &&
                e.start_time.getMonth() === nextEvent.start_time.getMonth() &&
                e.start_time.getDate() === nextEvent.start_time.getDate()
            );
            const formatterDate = new Intl.DateTimeFormat('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
            selectDay(document.querySelector(`.ohm-day[data-date="${nextEvent.start_time.getFullYear()}-${nextEvent.start_time.getMonth()}-${nextEvent.start_time.getDate()}"]`), formatterDate.format(nextEvent.start_time), dayEvents);
        } else {
            if (events.length > 0) {
                const firstEvent = [...events].sort((a,b) => a.start_time - b.start_time)[0];
                const dayEvents = events.filter(e => 
                    e.start_time.getFullYear() === firstEvent.start_time.getFullYear() &&
                    e.start_time.getMonth() === firstEvent.start_time.getMonth() &&
                    e.start_time.getDate() === firstEvent.start_time.getDate()
                );
                const formatterDate = new Intl.DateTimeFormat('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
                selectDay(document.querySelector(`.ohm-day[data-date="${firstEvent.start_time.getFullYear()}-${firstEvent.start_time.getMonth()}-${firstEvent.start_time.getDate()}"]`), formatterDate.format(firstEvent.start_time), dayEvents);
            } else {
                selectDay(null, "Upcoming Events", []);
                detailsContent.innerHTML = '<div class="ohm-empty-state">No events this month</div>';
            }
        }
    }
    selectUpcomingEvent();
}

    const detailsPanel = document.getElementById('ohm-calendar-details');
    const detailsClose = document.getElementById('ohm-details-close');

        detailsPanel.classList.remove('active');

    function selectDay(dayEl, title, dayEvents) {
        document.querySelectorAll('.ohm-day.selected').forEach(el => el.classList.remove('selected'));
        if (dayEl) dayEl.classList.add('selected');
        displayEvents(title, dayEvents);
    }
    
    function displayEvents(title, dayEvents) {
        detailsPanel.classList.add('active');
        detailsTitle.textContent = title;
        detailsContent.innerHTML = '';
        
        if (!dayEvents || dayEvents.length === 0) {
            detailsContent.innerHTML = '<div class="ohm-empty-state">No events found.</div>';
            return;
        }

        const formatterDate = new Intl.DateTimeFormat('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
        const formatterTime = new Intl.DateTimeFormat('en-US', { hour: 'numeric', minute: '2-digit' });
        
        dayEvents.forEach(event => {
            const dateStr = formatterDate.format(event.start_time);
            const startTimeStr = formatterTime.format(event.start_time);
            let timeString = `${dateStr} • ${startTimeStr}`;
            
            if (event.end_time) {
                const endTimeStr = formatterTime.format(new Date(event.end_time));
                timeString += ` - ${endTimeStr}`;
            }

            let locationString = event.location_name || '';
            if (event.address) {
                 locationString += locationString ? `<br><span style="color:#888; font-size:12px;">${event.address}</span>` : event.address;
            }

            const card = document.createElement('div');
            card.className = 'ohm-card';
            card.innerHTML = `
                ${event.image_url ? `<img src="${event.image_url}" alt="${event.title}" class="ohm-card-img" loading="lazy">` : ''}
                <div class="ohm-card-content">
                    ${event.tag ? `<div class="ohm-card-tag">${event.tag}</div>` : ''}
                    <h3 class="ohm-card-title">${event.title}</h3>
                    <div class="ohm-card-meta">
                        <svg viewBox="0 0 24 24"><path d="M19 4h-1V2h-2v2H8V2H6v2H5c-1.11 0-1.99.9-1.99 2L3 20c0 1.1.89 2 2 2h14c1.1 0 2-.9 2-2V6c0-1.1-.9-2-2-2zm0 16H5V10h14v10zM9 14H7v-2h2v2zm4 0h-2v-2h2v2zm4 0h-2v-2h2v2zm-8 4H7v-2h2v2zm4 0h-2v-2h2v2zm4 0h-2v-2h2v2z"/></svg>
                        <div>${timeString}</div>
                    </div>
                    ${locationString ? `
                    <div class="ohm-card-meta">
                        <svg viewBox="0 0 24 24"><path d="M12 2C8.13 2 5 5.13 5 9c0 5.25 7 13 7 13s7-7.75 7-13c0-3.87-3.13-7-7-7zm0 9.5c-1.38 0-2.5-1.12-2.5-2.5s1.12-2.5 2.5-2.5 2.5 1.12 2.5 2.5-1.12 2.5-2.5 2.5z"/></svg>
                        <div>${locationString}</div>
                    </div>` : ''}
                    ${event.description ? `<div class="ohm-card-desc">${event.description}</div>` : '<div style="flex-grow:1;"></div>'}
                    ${event.ticket_url ? `
                    <a href="${event.ticket_url}" target="_blank" rel="noopener noreferrer" class="ohm-ticket-btn">
                        Get Tickets
                        <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><line x1="5" y1="12" x2="19" y2="12"></line><polyline points="12 5 19 12 12 19"></polyline></svg>
                    </a>
                    ` : ''}
                </div>
            `;
            detailsContent.appendChild(card);
        });
        
        
    }

    // Init
    fetchEvents();
})();

    function applyFilters() {
        if (allEvents.length === 0) { renderCards([]); return; }

        let filtered = allEvents;
        const now = new Date();
        
        // Date Filter constraints
        if (currentFilter === 'upcoming') {
            filtered = filtered.filter(e => new Date(e.start_time) >= now);
        } else if (currentFilter === 'today') {
            const todayEnd = new Date(now);
            todayEnd.setHours(23, 59, 59, 999);
            const todayStart = new Date(now);
            todayStart.setHours(0,0,0,0);
            filtered = filtered.filter(e => {
                const start = new Date(e.start_time);
                return start >= todayStart && start <= todayEnd;
            });
        } 
        // 'all' includes everything fetched (which is from todayStart onwards)

        // Search Filter constraint
        if (searchQuery) {
            filtered = filtered.filter(e => {
                const q = searchQuery;
                const matchTitle = e.title && e.title.toLowerCase().includes(q);
                const matchLocation = e.location_name && e.location_name.toLowerCase().includes(q);
                const matchDesc = e.description && e.description.toLowerCase().includes(q);
                return matchTitle || matchLocation || matchDesc;
            });
        }

        renderCards(filtered);
    }

    function renderCards(events) {
        grid.innerHTML = '';
        countEl.textContent = `${events.length} event${events.length === 1 ? '' : 's'}`;

        if (events.length === 0) {
            grid.innerHTML = `<div class="ohm-state-msg"><p>No upcoming events found.</p></div>`;
            return;
        }

        const formatterDate = new Intl.DateTimeFormat('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
        const formatterTime = new Intl.DateTimeFormat('en-US', { hour: 'numeric', minute: '2-digit' });

        events.forEach(event => {
            const card = document.createElement('div');
            card.className = 'ohm-card';
            
            const dateStr = formatterDate.format(new Date(event.start_time));
            const startTimeStr = formatterTime.format(new Date(event.start_time));
            let timeString = `${dateStr} • ${startTimeStr}`;
            
            if (event.end_time) {
                const endTimeStr = formatterTime.format(new Date(event.end_time));
                timeString += ` - ${endTimeStr}`;
            }

            // Combine location name and address if both exist
            let locationString = event.location_name || '';
            if (event.address) {
                 locationString += locationString ? `<br><span style="color:#888; font-size:12px;">${event.address}</span>` : event.address;
            }

            card.innerHTML = `
                ${event.image_url ? `<img src="${event.image_url}" alt="${event.title}" class="ohm-card-img" loading="lazy">` : `<div class="ohm-card-img"></div>`}
                <div class="ohm-card-content">
                    ${event.tag ? `<div class="ohm-card-tag">${event.tag}</div>` : ''}
                    <h3 class="ohm-card-title">${event.title}</h3>
                    
                    <div class="ohm-card-meta">
                        <svg viewBox="0 0 24 24"><path d="M19 4h-1V2h-2v2H8V2H6v2H5c-1.11 0-1.99.9-1.99 2L3 20c0 1.1.89 2 2 2h14c1.1 0 2-.9 2-2V6c0-1.1-.9-2-2-2zm0 16H5V10h14v10zM9 14H7v-2h2v2zm4 0h-2v-2h2v2zm4 0h-2v-2h2v2zm-8 4H7v-2h2v2zm4 0h-2v-2h2v2zm4 0h-2v-2h2v2z"/></svg>
                        <div>${timeString}</div>
                    </div>
                    
                    ${locationString ? `
                    <div class="ohm-card-meta">
                        <svg viewBox="0 0 24 24"><path d="M12 2C8.13 2 5 5.13 5 9c0 5.25 7 13 7 13s7-7.75 7-13c0-3.87-3.13-7-7-7zm0 9.5c-1.38 0-2.5-1.12-2.5-2.5s1.12-2.5 2.5-2.5 2.5 1.12 2.5 2.5-1.12 2.5-2.5 2.5z"/></svg>
                        <div>${locationString}</div>
                    </div>` : ''}
                    
                    ${event.description ? `<div class="ohm-card-desc">${event.description}</div>` : '<div style="flex-grow:1;"></div>'}
                    
                    ${event.ticket_url ? `
                    <a href="${event.ticket_url}" target="_blank" rel="noopener noreferrer" class="ohm-ticket-btn">
                        Get Tickets
                        <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><line x1="5" y1="12" x2="19" y2="12"></line><polyline points="12 5 19 12 12 19"></polyline></svg>
                    </a>
                    ` : ''}
                </div>
            `;
            grid.appendChild(card);
        });
    }

    fetchEvents();
})();
